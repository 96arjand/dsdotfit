```python {{ language: 'python' }}
from linkedIn import JobDatabase

class Candidate:
    def __init__(self, skills, level):
        self.skills = skills
        self.level = level

    def fit(self, jobs=JobDatabase):
        offers = [job for job in jobs if job.matches(self)]
        return offers

candidate = Candidate(
    skills=['Python', 'Stats'],
    level='noob'
)

candidate.fit()
```

```sql {{ language: 'sql' }}
SELECT *
FROM candidates
WHERE
    (skills LIKE '%Python%' AND skills LIKE '%Stats%')
    AND level = 'noob'
```

```r {{ language: 'r' }}
# library(linkedIn)

Candidate <- R6Class(
  "Candidate",

  public = list(
    skills = NULL,
    level = NULL,

    initialize = function(skills, level) {
      self$skills <- skills
      self$level <- level
    },

    fit = function(jobs = list()) {
      offers <- lapply(jobs, function(job) {
        if(job$matches(self)) {
          return(job)
        } else {
          return(NULL)
        }
      })
      offers <- offers[!sapply(offers, is.null)]
      return(offers)
    }
  )
)

candidate <- Candidate$new(
  skills = c('Python', 'Stats'),
  level = 'noob'
)

offers_for_candidate <- candidate$fit()
```
